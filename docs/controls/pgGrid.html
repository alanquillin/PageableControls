pgGrid

A pageable and sortable enhanced table control.

  • Basic Grid Example
  • Column formatter

1. Basic Grid Example

Code:

<div id="myGridDiv"></div>

<script language="JavaScript">
    $(document).ready(function () {
        $('#myGridDiv').pgGrid({
            url: '/support/grid-data.php',
            dataModel: [
                { name: 'Id', index: 'id'},
                { name: 'Value', index: 'value'}
            ],
            title: "<h3>Test Data</h3>"
        });
    });
</script>

2. Column formatter

Code:

<div id="myGridDiv2"></div>

<script language="JavaScript">
    $(document).ready(function () {
        $('#myGridDiv2').pgGrid({
            url: '/support/grid-data.php',
            dataModel: [
                { name: 'Id', index: 'id'},
                { name: 'Value', index: 'value',
                    formatter: function(val, item){
                        if(Math.floor(item.id) % 2 == 0)
                            return val;

                        return val + ' [edited by formatter]'
                    }
                },
                { name: 'Fibonacci Sequence', sortable: false,
                    formatter: function(val, item){
                        var first,second,seq;
                        for(var i=0;i<item.id;i++){
                            if(i === 0){
                                first = 1;
                                second = 2;
                            }
                            seq = first + second;
                            first = second;
                            second = seq;
                        }
                        return seq
                    }
                }
            ],
            title: "<h3>Test Data</h3>"
        });
    });
</script>
