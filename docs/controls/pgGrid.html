Getting started

To start you need to add an element to the markup (normally a <div />).

1. Using the JQuery plugin method (recommended) 
<div id="myGridDiv"></div>

<script language="JavaScript">
    $(document).ready(function () {
        $('#myGridDiv').pgGrid('/support/grid-data.php',
        [
            { name: 'Id', index: 'id'},
            { name: 'Value', index: 'value'}
        ], {});
    });
</script>
2. Using the prototype method 
<div id="myGridDiv"></div>

<script language="JavaScript">
    $(document).ready(function () {
        var grid = new PGGrid($('#myGridDiv'), '/support/grid-data.php',
        [
            { name: 'Id', index: 'id'},
            { name: 'Value', index: 'value'}
        ],{});
    });
</script>

Parameters

Data Model Array<Object> Defines how the data will be displayed and formatted in the gid. Details

2. Data formatter

Code:

<div id="myGridDiv2"></div>

<script language="JavaScript">
    $(document).ready(function () {
        $('#myGridDiv2').pgGrid('/support/grid-data.php',
        [
            { name: 'Id', index: 'id'},
            { name: 'Value', index: 'value',
                dataFormatter: function(val, item){
                    if(Math.floor(item.id) % 2 == 0)
                        return val;

                    return val + ' [edited by formatter]'
                }
            },
            { name: 'Fibonacci Sequence', sortable: false,
                dataFormatter: function(val, item){
                    var first,second,seq;
                    for(var i=0;i<item.id;i++){
                        if(i === 0){
                            first = 1;
                            second = 2;
                        }
                        seq = first + second;
                        first = second;
                        second = seq;
                    }
                    return seq
                }
            }
        ],
        {
            title: "<h3>Test Data</h3>"
        });
    });
</script>

3. Cell formatter

Code:

<div id="myGridDiv3"></div>

<script language="JavaScript">
    $(document).ready(function () {
        $('#myGridDiv3').pgGrid('/support/grid-data.php',
        [
            { name: 'Id', index: 'id'},
            { name: 'Value', index: 'value', cellFormatter: function(cellValue, item, $cell){
                if(parseInt(item.id) % 2 == 0){
                    $cell.addClass('info')
                }
            }}
        ],
        {
            title: "<h3>Test Data</h3>"
        });
    });
</script>
