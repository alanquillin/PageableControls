1. Basic Grid Example

    $(document).ready(function () {
        $('#myGridDiv').pgGrid({
            url: '/support/grid-data.php',

                { name: 'Value', index: 'value'}
            ],
            title: "<h3>Test Data</h3>"
        });
    });
</script>

2. Data formatter

Code:

<div id="myGridDiv2"></div>

<script language="JavaScript">
    $(document).ready(function () {
        $('#myGridDiv2').pgGrid({
            url: '/support/grid-data.php',
            dataModel: [
                { name: 'Id', index: 'id'},
                { name: 'Value', index: 'value',
                    dataFormatter: function(val, item){
                        if(Math.floor(item.id) % 2 == 0)
                            return val;

                        return val + ' [edited by formatter]'
                    }
                },
                { name: 'Fibonacci Sequence', sortable: false,
                    dataFormatter: function(val, item){
                        var first,second,seq;
                        for(var i=0;i<item.id;i++){
                            if(i === 0){
                                first = 1;
                                second = 2;
                            }
                            seq = first + second;
                            first = second;
                            second = seq;
                        }
                        return seq
                    }
                }
            ],
            title: "<h3>Test Data</h3>"
        });
    });
</script>

3. Cell formatter

Code:

<div id="myGridDiv3"></div>

<script language="JavaScript">
    $(document).ready(function () {
        $('#myGridDiv3').pgGrid({
            url: '/support/grid-data.php',
            dataModel: [
                { name: 'Id', index: 'id'},
                { name: 'Value', index: 'value', cellFormatter: function(cellValue, item, $cell){
                    if(parseInt(item.id) % 2 == 0){
                        $cell.addClass('info')
                    }
                }}
            ],
            title: "<h3>Test Data</h3>"
        });
    });
</script>

4. Row formatter

Code:

<div id="myGridDiv4"></div>

<script language="JavaScript">
    $(document).ready(function () {
        $('#myGridDiv4').pgGrid({
            url: '/support/grid-data.php',
            rowFormatter: function($row, item){
                if(parseInt(item.id) % 2 == 0){
                    $row.css('background-color', 'red')
                }
            },
            dataModel: [
                { name: 'Id', index: 'id'},
                { name: 'Value', index: 'value'}
            ],
            title: "<h3>Test Data</h3>"
        });
    });
</script>
